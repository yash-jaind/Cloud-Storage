const multer = require('multer');
const storage = multer.memoryStorage(); // multer now stores files in memory as Buffer
const upload = multer({ storage: storage });

  //In Node.js (using Multer for file uploads),
memoryStorage() means the uploaded files are not saved on your computer’s disk — instead, they are temporarily kept in the server’s RAM (memory).
//only use memoryStorage() for small files or temporary use,
otherwise, it can crash your server if too many large files are uploaded.
// will tell later how to store permanently in this notepad
//How it works step by step:

When a user uploads a file (for example, a photo or document),

Multer takes that uploaded file and stores it directly in memory (RAM),
instead of saving it to a folder like uploads/.
//Step 2: What happens next

At this stage, the file is not stored permanently — it only exists in memory until your code saves it somewhere.

If you do nothing, the data is lost after the request finishes.
//When you use memoryStorage(), Multer gives you a req.file object like this:
//{
  fieldname: 'file',
  originalname: 'photo.jpg',
  encoding: '7bit',
  mimetype: 'image/jpeg',
  buffer: <Buffer ff d8 ff e0 00 10 4a 46 ... >, // <-- actual file data in memory
  size: 12845
}//
//req.file.buffer

This property (req.file.buffer) contains the entire file data as a binary buffer.

You can use it directly in code — for example:

Upload it to cloud storage (like AWS S3, Google Cloud, or Firebase).

Convert it to a base64 string.

Or process it (resize, compress, etc.) using libraries like sharp.

const fileSchema = new mongoose.Schema({
  filename: String,
  originalname: String,
  email: String,
  data: Buffer,          // Store file binary data here
  contentType: String,   // Store file MIME type
  uploadDate: { type: Date, default: Date.now },
   fileHash: String // Add this line!
});
const File = mongoose.model('File', fileSchema);



app.post('/api/signup', async (req, res) => {
  try {
    const { email, password } = req.body;
    const existingUser = await User.findOne({ email });
    if (existingUser) {
      return res.status(400).json({ error: "User already exists" });
    }
    const hash = await bcrypt.hash(password, 10);
    const user = new User({ email, password: hash });
    await user.save();
    res.json({ message: 'Signup successful' });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: 'Internal server error' });
  }
});



app.post('/api/login', async (req, res) => {
  const { email, password } = req.body;
  const user = await User.findOne({ email });
  if (!user) return res.status(401).json({ error: 'User not found' });
  const match = await bcrypt.compare(password, user.password);
  if (!match) return res.status(401).json({ error: 'Wrong password' });
  res.json({ message: 'Login successful' });
});


const crypto = require('crypto'); // Add this import


app.post('/api/upload', upload.single('file'), async (req, res) => {
  const { userEmail } = req.body;
  if (!req.file) {
    return res.status(400).json({ error: 'No file uploaded' });
  }
  try {
    // Create hash from file buffer
    const hash = crypto.createHash('sha256').update(req.file.buffer).digest('hex');
    
    // Check if file with same hash already exists for this user
    const existing = await File.findOne({ fileHash: hash, email: userEmail });
    if (existing) {
      return res.status(409).json({ error: 'Duplicate file detected. This file already exists.' });
    }


    const file = new File({
      filename: req.file.originalname,
      originalname: req.file.originalname,
      email: userEmail,
      data: req.file.buffer,
      contentType: req.file.mimetype,
      fileHash: hash // Save hash in document
    });
    await file.save();
    res.json({ message: 'File uploaded and saved to DB!' });
  } catch (err) {
    res.status(500).json({ error: 'Error saving file info' });
  }
});

app.get('/api/file/:id', async (req, res) => {
  try {
    const file = await File.findById(req.params.id);
    if (!file) return res.status(404).send('File not found');

    const isDownload = req.query.download === '1';
    res.setHeader('Content-Type', file.contentType);

    if (isDownload) {
      // Force download
      res.setHeader(
        'Content-Disposition',
        `attachment; filename="${file.originalname}"`
      );
    } else {
      // View inline (so browser displays it)
      res.setHeader(
        'Content-Disposition',
        `inline; filename="${file.originalname}"`
      );
    }

    res.send(file.data);
  } catch (err) {
    console.error(err);
    res.status(500).send('Error serving file');
  }
});
